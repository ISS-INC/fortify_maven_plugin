211 	public void execute() throws MojoExecutionException, MojoFailureException {
212 		if (!scanEnabled) {
213 			getLog().info("Scan disabled for this project. Skipping.");
214 			return;
215 		}
216 
217 		init();
218 
219 		if (!packaging.equals("pom") && !packaging.equals("jar") && !packaging.equals("war") && !packaging.equals("ear") && !packaging.equals("ejb") && !packaging.equals("maven-plugin") && !packaging.equals("bundle")) {
220 			getLog().info("******* Skipping scan for " + project.getArtifactId() + ". The SCA maven-plugin does not support the packaging value " + packaging + " ********");
221 			return;
222 		}
223 
224 		getLog().info("                   Packaging -> " + packaging);
225 		getLog().info("       Top-Level Artifact ID -> " + toplevelArtifactId);
226 		getLog().info("                 Build Label -> " + buildLabel);
227 		getLog().info("               Build Version -> " + buildVersion);

284 			throws MojoExecutionException, MojoFailureException {
285 		StreamConsumer output = new DefaultConsumer();
286 		StreamConsumer error = new DefaultConsumer();
287 		Commandline cli = new Commandline();

292 
293 		try {
294 			if (fortifyclient == null)

311 			} else {

315 				cli.createArgument().setValue(projectVersion);

319 			cli.createArgument().setValue(f360AuthToken);
320

323 
324 			getLog().info("cmd: \"" + cli.toString() + "\"");
325 
326 			// execute the command

328 			int result = CommandLineUtils.executeCommandLine(cli, output, error);
329 			getLog().debug("Exit code " + result);
330 			if (result != 0) {

338 									+ "\nPlease check your project settings, , and sourceanalyzer installation / invokation."

340 				}

346 			getLog().error("Please confirm that the fortifyclient executable is in the path or specified correctly using the fortifyclient configuration"
347 							+ " option.  Currently it is set to \""
348 							+ sourceanalyzer + "\"");
349 			throw new MojoExecutionException(e.getLocalizedMessage());
350 		} catch (Exception e) {
351 			getLog().error("While setting system environment to cli - " + e.getLocalizedMessage(), e);
352 			throw new MojoExecutionException(e.getLocalizedMessage());

356 	private String getProjectId(String projectName, String projectVersion)
357 			throws MojoExecutionException, MojoFailureException {
358 		StringStreamConsumer output = new StringStreamConsumer();
359 		StringStreamConsumer error = new StringStreamConsumer();
360 		Commandline cli = new Commandline();
361 		String projectId = null;
362 		try {
363 			if (fortifyclient == null)
364 				throw new IllegalArgumentException("No fortifyclient executable has been defined.  "
365 								+ "Please rectify this, if problems persist contact Fortify Professional Services for assistance.");
366 			cli.setExecutable(fortifyclient);
367 			cli.addSystemEnvironment();
368 
369 			cli.createArgument().setValue("listprojects");
370 			cli.createArgument().setValue("-url");
371 			cli.createArgument().setValue(f360Url);
372 			cli.createArgument().setValue("-authtoken");
373 			cli.createArgument().setValue(f360AuthToken);
374 
375 			getLog().info("cmd: \"" + cli.toString() + "\"");
376 
377 			// execute the command
378 			getLog().debug("Executing system command: " + cli.toString());
379 			int result = CommandLineUtils.executeCommandLine(cli, output, error);
380 			getLog().debug("Exit code " + result);
381 			if (result != 0) {
382 				getLog().error("The fortifyclient invocation failed. ResultCode: "
383 								+ result
384 								+ "\nPlease check your project settings, and sourceanalyzer installation / invokation."
385 								+ "\nContact Fortify Professional Services if this error persists.");
386 				if (failOnUploadError) {
387 					throw new MojoExecutionException("The fortifyclient invocation failed. ResultCode: "
388 									+ result
389 									+ "\nPlease check your project settings, , and sourceanalyzer installation / invokation."
390 									+ "\nContact Fortify Professional Services if this error persists.");
391 				}
392 			} else {
393 				// try to parse result and look for our specific project name
394 				String strOutput = output.getOutput();
395 				String regexp = ".*(\\d+)\\s+" + escapeRegexpChars(projectName) + "\\s+" + escapeRegexpChars(projectVersion) + ".*";
396 				getLog().debug("\n*******************\n"
397 								+ "******* fortifyclient OUPUT looking for Project: \""
398 								+ projectName + "\" Version: \""
399 								+ projectVersion + "\" ******* \n"
400 								+ "******* Regexp: \"" + regexp + "\"\n"
401 								+ "*******************\n" + strOutput
402 								+ "***** END OUTPUT *****\n");
403 				Pattern p = Pattern.compile(regexp, Pattern.MULTILINE | Pattern.DOTALL);
404 				Matcher m = p.matcher(strOutput);
405 				if (m.matches()) {
406 					projectId = m.group(1);
407 					getLog().info("** Found FM Project Id: " + projectId + " **");
408 				} else {
409 					getLog().info("** Project: " + projectName + " does not exist in SSC.  Skipping upload");
410 				}
411 			}
412 
413 			return projectId;
414 		} catch (CommandLineException e) {
415 			getLog().error("While executing command " + e, e);
416 			getLog().error("Please confirm that the fortifyclient executable is in the path or specified correctly using the fortifyclient configuration"
417 							+ " option.  Currently it is set to \""
418 							+ sourceanalyzer + "\"");
419 			throw new MojoExecutionException(e.getLocalizedMessage());
420 		} catch (Exception e) {
421 			getLog().error("While setting system environment to cli - " + e.getLocalizedMessage(), e);
422 			throw new MojoExecutionException(e.getLocalizedMessage());
423 		}
424 	}

463 	protected void validateSSCUploadArgs(String projectId, String projectName, String projectVersion, String f360Url, String f360AuthToken)
464 			throws MojoExecutionException {
465 
466 		// nothing to check if no upload
467 		if (!upload){
468 			return;
469 		}
470 
471 		StringBuffer errMsg = new StringBuffer();
472 
473 		// if upload then url, authtoken are mandatory
474 		if (!StringHelper.isDefinedNonTrivially(f360Url)) {
475 			errMsg.append("Please supply the Software Security Center url.\n");
476 		}
477 
478 		if (!StringHelper.isDefinedNonTrivially(f360AuthToken)) {
479 			errMsg.append("Please supply the Software Security Center authentication token.\n");
480 		}
481 
482 		// if projectId supplied then warn that it overrides projectName/Version
483 		// if supplied
484 		if (!StringHelper.isDefinedNonTrivially(projectId)) {
485 			if (projectName != null || projectVersion != null) {
486 				getLog().warn("Supplied upload projectId: " 
487 								+ projectId
488 								+ " overrides projectName/projectVersion combination: \""
489 								+ projectName + "\"/\"" + projectVersion + "\"");
490 			}
491 		} else {
492 			// no projectID then name/version required
493 			if (!StringHelper.isDefinedNonTrivially(projectName)) {
494 				errMsg.append("Please supply a valid projectName value.\n");
495 			}
496 
497 			if (!StringHelper.isDefinedNonTrivially(projectVersion)) {
498 				errMsg.append("Please supply a valid projectVersion value.\n");
499 			}
500 		}
501 
502 		if (errMsg.length()>0){
503 			throw new MojoExecutionException(errMsg.toString());
504 		}
505 	}
