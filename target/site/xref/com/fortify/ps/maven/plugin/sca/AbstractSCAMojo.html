256 	protected void addProperties(Properties props) {
257 		Enumeration keys = props.keys();
258 		while (keys.hasMoreElements()) {
259 			String currKey = (String) keys.nextElement();
260 			String option = "-D" + currKey + "=" + props.getProperty(currKey);
261 			addArg(option);
262 		}

265 	// Read in a .properties file and add each constituent property-value pair
266 	// to the command line in the form of
267 	// -Dproperty=value.
268 	protected void processPropertiesFile() throws MojoExecutionException {
269 		if (scaPropertiesFile == null) {
270 			return;
271 		}
272 
273 		if (!scaPropertiesFile.exists()) {
274 			throw new MojoExecutionException("Properties file " + scaPropertiesFile.getAbsolutePath() + " does not exist");
275 		}else if (scaPropertiesFile.isDirectory()) {
276 			throw new MojoExecutionException("Properties file " + scaPropertiesFile.getAbsolutePath() + " is a directory.");
277 		}
278 
279 		FileInputStream fis = null;
280 
281 		try {
282 			Properties scaRunTimeProps = new Properties();
283 			fis = new FileInputStream(scaPropertiesFile);
284 
285 			scaRunTimeProps.load(fis);
286 			addProperties(scaRunTimeProps);
287 		}catch (IOException x) {
288 			getLog().error("Failed to process " + scaPropertiesFile.getAbsolutePath());
289 
290 			throw new MojoExecutionException("Failed to process " + scaPropertiesFile.getAbsolutePath(), x.getCause());
291 		}finally {
292 			FileHelper.closeQuietly(fis);
293 		}
294 	}

310 		FileWriter fw = null;
311 		try {
312 			File argfile = new File(fileName);
313 			File parentDir = argfile.getParentFile();
314 			if (!parentDir.exists() && !parentDir.mkdirs()) {
315 				getLog().info("Failed to create dir " + parentDir.getAbsolutePath());
316 
317 				throw new MojoExecutionException("Unable to find / create specified output directory " + parentDir.getAbsolutePath());
318 			} else {
319 				getLog().info("Created output dir " + parentDir.getAbsolutePath());
320 			}
321 
322 			fw = new FileWriter(fileName);
323 			fw.write(argFileStrBuf.toString());
324 			fw.flush();
325 		} catch (IOException x) {
326 			getLog().error("Failed while attempting to write to " + fileName + " - " + x.getMessage());
327 			throw new MojoExecutionException(x.getMessage());
328 		} finally {
329 			FileHelper.closeQuietly(fw);
330 		}
331 	}

333 	/**
334 	 * @param argFile
335 	 *            for SCA invocation.
336 	 * @param failOnSCAError
337 	 *            when true, throws an exception if command-fails.
338 	 * @return true if scan was successful, false otherwise.
339 	 * @throws MojoExecutionException
340 	 * @throws MojoFailureException
341 	 */
342 	protected boolean invoke(String argFile, boolean failOnSCAError)
343 			throws MojoExecutionException, MojoFailureException {
344 		if (dontRunSCA) {
345 			getLog().info("dontRunSCA is set by the user. Collect the argument file from the target directory.");
346 			return true;
347 		}
348 
349 		StreamConsumer output = new DefaultConsumer();
350 		StreamConsumer error = new DefaultConsumer();
351 		Commandline cli = new Commandline();
352 		try {
353 			if (sourceanalyzer == null)
354 				throw new IllegalArgumentException("No sourceanalyzer executable has been defined.  " 
355 					+ "Please rectify this, if problems persist contact Fortify Professional Services for assistance.");
356 			cli.setExecutable(sourceanalyzer);
357 			cli.addSystemEnvironment();
358 
359 			cli.createArgument().setLine(commandStrBuf.toString());
360 			// -Xmx is a JVM argument. Hence, it must go outside the argument
361 			// file: By the time SCA starts up and parses
362 			// the argument file, it is too late -- the JVM has already started.
363 			if (maxHeap != null && maxHeap.length() > 0) {
364 				cli.createArgument().setLine("-Xmx" + maxHeap);
365 			}
366 			
367 			if (minHeap != null && minHeap.length() > 0 ) {
368 				cli.createArgument().setLine("-Xms" + minHeap);
369 			}
370 			
371 			if (maxPermGen != null && maxPermGen.length() > 0 ) {
372 				cli.createArgument().setLine("-XX:MaxPermSize=" + maxPermGen);
373 			}
374 			
375 			if (stackSize != null && stackSize.length() > 0 ) {
376 				cli.createArgument().setLine("-Xss" + stackSize);
377 			}
378 			cli.createArgument().setValue("@" + argFile);
379 			getLog().info("cmd: \"" + cli.toString() + "\"");
380 
381 			// execute the command
382 			getLog().debug("Executing system command: " + cli.toString());
383 			int result = CommandLineUtils
384 					.executeCommandLine(cli, output, error);
385 			getLog().debug("Exit code " + result);
386 			if (result != 0) {
387 				String errorMsg = "Error invoking sourceanalyzer. Exit code: "
388 								+ result 
389 								+ ".\nVerify your project settings and your SCA installation.";
390 
391 				getLog().error(errorMsg);
392 
393 				if (failOnSCAError) {
394 					throw new MojoExecutionException(errorMsg);
395 				}
396 			}
397 			return (result == 0);
398 		} catch (CommandLineException e) {
399 			getLog().error("While executing command " + e, e);
400 			getLog().error("Please confirm that the sourceanalyzer executable is in the path or specified correctly using the sourceanalyzer configuration"
401 							+ " option.  Currently it is set to \""
402 							+ sourceanalyzer + "\"");
403 			throw new MojoExecutionException(e.getLocalizedMessage());
404 		} catch (Exception e) {
405 			getLog().error("While setting system environment to cli - " + e.getLocalizedMessage(), e);
406 			throw new MojoExecutionException(e.getLocalizedMessage());
407 		}
408 	}
