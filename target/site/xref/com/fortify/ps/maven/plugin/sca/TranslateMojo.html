248 	/**
249 	 * Performs the translation step of an SCA analysis.
250 	 *
251 	 * @throws MojoExecutionException
252 	 * @throws MojoFailureException
253 	 */
254 	public void execute() throws MojoExecutionException, MojoFailureException {
255 		if (compileSourceRoots == null) {
256 			compileSourceRoots = project.getCompileSourceRoots();
257 		}
258 		if (!scanEnabled) {
259 			getLog().info("Scan disabled for this project. Skipping.");
260 			return;
261 		}
262 
263 		init();
264         compileJavaSource();
265 		identifyJavaSource();
266 
267 		if (packaging.equals("pom")) {
268 		    getLog().info("Packaging is POM. Skipping since its modules will be translated.");
269 		} else if (packaging.equals("jar") || packaging.equals("war") || packaging.equals("ear") || packaging.equals("ejb") || packaging.equals("maven-plugin") || packaging.equals("bundle")){
270 			logBasicInfo();
271 			translateJavaSources(); // Always translate the Java sources in a project.
272 			if (packaging.equals("war")) {
273 				translateWar();
274 			} else if (packaging.equals("ear"))  {
275 				translateEar();
276 			}
277 		} else {
278 			getLog().info("******* The SCA plug-in does not support the packaging value " + packaging + " ********");
279 		}
280 		projectArtifact.setFile( classesDirectory );
281 	}

283 	private void compileJavaSource() throws MojoExecutionException {
284 		
285 		if (!classesDirectory.exists() || !FileHelper.containsClassFiles(classesDirectory)) {
286 			Plugin compilerPlugin = null;
287 			
288 			for( Plugin plug : plugins) {
289 				if (plug.getGroupId().equals("org.apache.maven.plugins") && plug.getArtifactId().equals("maven-compiler-plugin")) {
290 					compilerPlugin = plug;
291 					break;
292 				}
293 			}
294 			if (compilerPlugin == null) {
295 				compilerPlugin = MojoExecutor.plugin(MojoExecutor.groupId("org.apache.maven.plugins"), MojoExecutor.artifactId("maven-compiler-plugin"), MojoExecutor.version("3.1"));
296 				plugins.add(compilerPlugin);
297 			}
298 			
299 			Xpp3Dom config = (Xpp3Dom)compilerPlugin.getConfiguration();
300 			
301 			if (config == null) {
302 				config = MojoExecutor.configuration(
303 					MojoExecutor.element(MojoExecutor.name("source"), source),
304 					MojoExecutor.element(MojoExecutor.name("target"), source)
305 				);
306 			}
307 			MojoExecutor.ExecutionEnvironment environment = MojoExecutor.executionEnvironment(project, session, pluginManager);
308 			MojoExecutor.executeMojo(compilerPlugin, MojoExecutor.goal("compile"), config, environment);
309 		}
310 	}

343 	/**
344 	 * Generate SCA arguments to translate the WAR represented by this project. The class path automatically includes
345 	 * WEB-INF/classes and the JARs inside WEB-INF/lib. If class path information was stored by a prior ear goal
346 	 * invocation, then this too is added. Additionally, the user may supply additional class path entries using the
347 	 * fortify.sca.cp command line property or extraClassPathString POM property. 
348 	 * 
349 	 * @throws MojoExecutionException
350 	 */
351 	protected void generateArgsForWar() throws MojoExecutionException {
352 		constructCommandLineArgs();
353 
354 		boolean useSourceDirectory = false;
355 		String warTargetRoot = warOutputDirectory.getAbsolutePath();
356 		String warSourceRoot = getWarSourceRoot();
357 
358 		if (buildId != null) {
359 			// If generated Java sources are available, use them.
360 
361 			// SYNTAX
362 			// sourceanalyzer -b bid -document-root main/webapp -generated-sources /path/to/generated/sources
363 			//                -cp ...
364 
365 			String generatedSourcesDir = util.showModule(buildId, project.getGroupId(), project.getArtifactId());
366 			if (StringHelper.isDefinedNonTrivially(generatedSourcesDir)) {
367 				getLog().info("Using available generated Java sources at " + generatedSourcesDir);
368 				addArg("-generated-sources");
369 				addArg(generatedSourcesDir);
370 				addArg("-document-root");
371 				addArg(warSourceRoot);
372 				useSourceDirectory = true;
373 			}
374 		}
375 
376 		if (!useSourceDirectory) {
377 			if (warOutputDirectory.exists()) {
378 				getLog().info("Generated sources not found. Using exploded WAR location at " + warOutputDirectory);
379 				addArg(warTargetRoot);
380 			} else {
381 				getLog().info("Target folder not found. Using source files at " + warSourceRoot);
382 				addArg(warSourceRoot);
383 			}
384 		}
385 
386 		// When building the class path, use the Web app root in the target directory rather than the one in the source
387 		// directory itself. Sometimes, needed class path entries do not appear until after compilation, the output of
388 		// which appears in the target directory.
389 
390 		// -cp <saved class path>;WEB-INF/classes;<extra class path>
391 		String warCP = warTargetRoot + File.separator + "WEB-INF" + File.separator + "classes";
392 		String combinedCP = StringHelper.concatenateClassPaths(warCP, extraClassPathString); // user-provided
393 		combinedCP = StringHelper.concatenateClassPaths(util.showClassPath(buildId), combinedCP); // stored by ear goal
394 		addOptionValuePair("-cp", combinedCP);
395 
396 		// -extdirs WEB-INF/lib
397 		addArg("-extdirs") ;
398 		addArg(warTargetRoot + File.separator + "WEB-INF" + File.separator + "lib");
399 	}

412 		String defaultLoc = new File(baseDir + File.separator + "src" + File.separator + "main" + File.separator + "application").getAbsolutePath();
413 		String earSourceDirectory = config.getElemValue("earSourceDirectory", defaultLoc);
414 		File earSourceDirectoryFile = new File(earSourceDirectory);
415 
416 		if (!earSourceDirectoryFile.exists()) {
417 			getLog().info("This EAR module does not contain any resources.");
418 			return;
419 		}

421 		constructCommandLineArgs();
422 		if (earSourceDirectoryFile.isAbsolute()) {
423 			addArg(earSourceDirectory);
424 		} else {
425 			addArg(new File(baseDir, earSourceDirectory).getAbsolutePath());
426 		}
427 
428 		writeArgFile(scaArgsFileNamePrefix + "ear.txt") ;
429 		invoke(scaArgsFileNamePrefix + "ear.txt", failOnSCAError) ;
430 	}

490 		getLog().info ("              Source Version -> " + (source == null ? "SCA default" : source) );
491 		getLog().info ("           Source File Paths -> " + StringHelper.listToString(compileSourceRoots, " "));
492 		getLog().info ("      Test Source File Paths -> " + StringHelper.listToString(testCompileSourceRoots, " "));
493         getLog().info ("         Excluded File Paths -> " + exclude);
494 
495 		boolean retVal = false;
496 		Iterator<String> it = compileSourceRoots.iterator();
497 		while (it.hasNext()) {
498 			String currSourceRoot = it.next();
499 			if (new File(currSourceRoot).isDirectory()) {
500 				addArg(currSourceRoot);
501 				retVal = true;
502 			}
503 		}
504 		if (!skipTests) {
505 			it = testCompileSourceRoots.iterator();
506 			while (it.hasNext()) {
507 				String currRoot = it.next();
508 				if (new File(currRoot).isDirectory()) {
509 					addArg(it.next());
510 					retVal = true;
511 				}
512 			}
513 		}
514 		return retVal;
515 	}

518 	 * Add the resources to the SCA build. Two important resources that need to be added are ejb-jar.xml and
519 	 * hibernate.properties. Otherwise, it is usually worth adding resources since they may contain issues such as
520 	 * Hardcoded Passwords.

522 	protected boolean addResources() {
523 		List<String> resourceStrList = getResourcesAsStrings(resources);
524 		getLog().info ("                   Resources -> " + StringHelper.listToString(resourceStrList, " "));
525 		Iterator<String> resIter = resourceStrList.iterator();
526 		while (resIter.hasNext()) {
527 			String currRes = resIter.next();
528 			if (new File(currRes).exists()) {
529 				addArg(currRes);
530 			}
531 		}
532 		List<String> testResourceStrList = null;
533 		if (!skipTests) {
534 			testResourceStrList = getResourcesAsStrings(testResources);
535 			getLog().info ("              Test Resources -> " + StringHelper.listToString(testResourceStrList, " "));
536 			resIter = testResourceStrList.iterator();
537 			while (resIter.hasNext()) {
538 				String currRes = resIter.next();
539 				if (new File(currRes).exists()) {
540 					addArg(currRes);
541 				}
542 			}
543 		}
544 		return resourceStrList.size() > 0 || (testResourceStrList != null ? testResourceStrList.size() > 0 : false);
545 	}

547 	/**
548 	 * Turns a List&lt;Resource&gt; to a List&lt;String&gt;. The return value is useful for feeding to listToString().
549 	 * 
550 	 * @param resources List to turn into a list of strings.
551 	 * @return A list of String objects.
552 	 * @see <a href="http://maven.apache.org/pom.html#Resources">Maven Resources</a>
553 	 */
554 	private List<String> getResourcesAsStrings(List<Resource> resources) {
555 		/*
556 			- resources:  is a list of resource elements that each describe what and where to include files associated with this project.
557 			- targetPath: Specifies the directory structure to place the set of resources from a build. Target path defaults to the base directory.
558 			              A commonly specified target path for resources that will be packaged in a JAR is META-INF.
559 			- filtering:  is true or false, denoting if filtering is to be enabled for this resource.
560 			- directory:  This element's value defines where the resources are to be found. The default directory for a build is ${basedir}/src/main/resources.
561 			- includes:   A set of files patterns which specify the files to include as resources under that specified directory, using * as a wildcard.
562 			- excludes:   The same structure as includes, but specifies which files to ignore. In conflicts between include and exclude, exclude wins.
563 		*/
564 		List<String> retVal = new ArrayList<String>();
565 		
566 		for( Resource rcs : resources) {
567 			if (rcs.isFiltering() == false ) {
568 				String dir = rcs.getDirectory();
569 				
570 				if( FileHelper.isDirectory(dir) ) {
571 					List<String> includes = rcs.getIncludes();
572 					List<String> excludes = rcs.getExcludes();
573 					if ( includes.size() == 0 && excludes.size() == 0 ) {
574 						retVal.add(dir);
575 					} else {
576 						List<String> filePaths = FileHelper.getAllFilesInDirectory(dir);
577 						if (includes.size() == 0 ) {
578 							retVal.addAll(filePaths);
579 						} else {
580 							for(String filePath : filePaths) {
581 								for( String include : includes) {
582 									File file = new File(filePath);
583 									File includePattern = new File(dir + File.separator + include);
584 									if ( matches(file, includePattern) ) {
585 										retVal.add(filePath);
586 									}
587 								}
588 							}
589 						}
590 						
591 						for(String filePath : filePaths) {
592 							for( String exclude : excludes) {
593 								File file = new File(filePath);
594 								File excludePattern = new File(dir + File.separator + exclude);
595 								if ( matches(file, excludePattern) ) {
596 									retVal.remove(filePath);
597 								}
598 							}
599 						}
600 					}
601 				}
602 			}
603 		}
604 		return retVal;
605 	}

607 	private boolean matches(File file, File pattern)
608 	{
609 		String filePath = StringHelper.convertFileSeparator(file.getAbsolutePath());
610 		String filePattern = StringHelper.convertFileSeparator(pattern.getAbsolutePath());
611 		filePattern = StringHelper.createRegexFromFilePattern(filePattern);
612 		
613 		Pattern p = Pattern.compile(filePattern);
614 		Matcher m = p.matcher(filePath);
615 		
616 		return m.matches();
617 	}

619 	private void identifyJavaSource(){
620 		Plugin plugin = util.getPlugin("org.apache.maven.plugins", "maven-compiler-plugin");
621 		Util.SafeConfig config = new Util.SafeConfig(plugin);
622 		String javacSource = config.getElemValue("source");
623 		
624 		if( source == null ) {
625 			if( javacSource != null ) {
626 				getLog().info("Translating java source as " + javacSource + ".");
627 				source = javacSource;
